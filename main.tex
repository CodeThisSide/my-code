% \entry{Title}{Dates}{Organisation}{Location}
% Renders two lines: bold title + italic org on left,
% dates + location on right. Location can be empty.
\newcommand{\entry}[4]{%
  \vspace{4pt}%
  \noindent
  \begin{minipage}[t]{0.71\linewidth}
    \textbf{#1}%
    \ifx\relax#3\relax\else\\[-1pt]\textit{\small #3}\fi
  \end{minipage}%
  \hfill
  \begin{minipage}[t]{0.27\linewidth}
    \raggedleft\small #2%
    \ifx\relax#4\relax\else\\[-1pt]\textit{\small #4}\fi
  \end{minipage}%
  \vspace{0pt}%
}

% \proj{Project Name}{Tech Stack}
\newcommand{\proj}[2]{%
  \vspace{4pt}%
  \noindent\textbf{#1}\hfill\textit{\small #2}\\[-4pt]%
}
